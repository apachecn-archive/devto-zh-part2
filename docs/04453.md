  let area = 1;
  for (let i = 1; i <= n; i++) {
    area += i * 4 - 4;
  }
  return area;

  if (n === 1) {
    return 1;
  } else {
    return n * 4 - 4 + x.recursionShapeArea(n - 1);

  return Math.pow(n, 2) + Math.pow(n - 1, 2);

const shapeArea = (n) => {
  let solution = {};

    let obj = {
      0: "loopShapeArea",
