 # Useful Function
function commiter() {
    # Add file, commit and push
    git add -f "$1";
    if [ "$2" == "" ]; then git commit -m"Updated $1";
    else git commit -m"$2";
    fi;
    $(git push -q >> /dev/null 2>&1) &
    }

function ssh() {
    # Always ssh with -X
    command ssh -X "$@"
}

function rsync(){
    command rsync --progress "$@"
}

function extract () {
     if [ -f $1 ] ; then
         case $1 in
             *.tar.bz2)   tar xjf $1    ;;
             *.tar.gz)    tar xzf $1    ;;
             *.bz2)       bunzip2 $1    ;;
             *.rar)       rar x $1      ;;
             *.gz)        gunzip $1     ;;
             *.tar)       tar xf $1     ;;
             *.tbz2)      tar xjf $1    ;;
             *.tgz)       tar xzf $1    ;;
             *.zip)       unzip $1      ;;
             *.Z)         uncompress $1 ;;
             *.7z)        7z x $1       ;;
             *)           echo "'$1' cannot be extracted via extract()" ;;
         esac
     else echo "'$1' is not a valid file"
     fi
}

function psgrep() {
    if [ ! -z $1 ] ; then echo "Grepping for processes matching $1..."
        ps aux | grep $1 | grep -v grep else echo "!! Need name to grep for"
    fi
}
